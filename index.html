import requests from bs4 import BeautifulSoup url = 'https://www.sephora.com/product/daisy-eau-de-toilette-P191854' response = requests.get(url) soup = BeautifulSoup(response.content, 'html.parser') price_elements = soup.find_all('span', class_='css-0') prices = [float(e.text.replace('$', '')) for e in price_elements] lowest_price = min(prices) print(f'The lowest price for Marc Jacobs Daisy perfume is ${lowest_price:.2f}')
